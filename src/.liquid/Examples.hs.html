 21: valid_parallel_thirds :: Counterpoint
 22: Counterpoint.Counterpointvalid_parallel_thirds =
 23:        Counterpoint.BarmakeBar {v : Counterpoint.Pitch | v == C}C GHC.Types.Int4 {v : Counterpoint.Pitch | v == E}E GHC.Types.Int4   -- 長3度 (4半音)

 25:     :| Counterpoint.BarmakeBar {v : Counterpoint.Pitch | v == E}E GHC.Types.Int4 {v : Counterpoint.Pitch | v == G}G GHC.Types.Int4   -- 短3度 (3半音)
 26:     :| Counterpoint.BarmakeBar {v : Counterpoint.Pitch | v == F}F GHC.Types.Int4 {v : Counterpoint.Pitch | v == A}A GHC.Types.Int4   -- 長3度 (4半音)
 27:     :| Counterpoint.BarmakeBar {v : Counterpoint.Pitch | v == G}G GHC.Types.Int4 {v : Counterpoint.Pitch | v == H}H GHC.Types.Int4   -- 長3度 (4半音)
